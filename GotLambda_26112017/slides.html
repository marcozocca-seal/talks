      MonadHttp m,

     method

     -> body

A GET request

request1 :: MonadHttp m => m LB.ByteString
request1 = do
   r <- req
      GET
      (https "www.google.com")
      NoReqBody
      lbsResponse
      mempty
   return $ responseBody r
